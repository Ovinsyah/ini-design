## Getting Started
```
npm install --save ini-design styled-components
```
```
yarn add ini-design styled-components
```
## Usage
```jsx
import { IniButton } from 'ini-design';

const App = () => {
  return (
    <>
      <IniButton theme="primary">Click Me</IniButton>
    </>
  )
}
```
